select ?person ?last ?based ?ipcountry ?bday  ?since ?gen 
  ((select group_concat (?lng, ",") where {?person sv:speaks ?lng} group by ?person))
  ((select group_concat ( bif:concat (?o_name, ?o_country), ",") where {?person sv:work_at ?w . ?w sv:organization ?org . ?org dbpedia-owl:country ?o_country . ?org foaf:name ?o_name } group by ?person))
{
?person a                                   sv:Person .
?person sv:firstName "%Name%" .
?person sv:based_near                       ?based .
?person sv:birthday                         ?bday .
?person sv:created                         ?since . 
?person sv:gender                         ?gen .
?person sv:lastName ?last .
} order by ?last limit 10
