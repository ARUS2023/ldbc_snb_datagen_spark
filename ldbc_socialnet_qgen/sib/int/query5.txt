where { {{ %Person% sv:knows ?fr} union {  %Person% sv:knows ?fr2 . ?fr2 sv:knows ?fr}}
  ?group sv:member ?mem . ?mem sv:person ?fr . ?mem sv:join ?date . filter (?date > "%Date0%"^^xsd:date) .
  ?post sv:creater ?fr . ?gr sv:container_of ?post }
