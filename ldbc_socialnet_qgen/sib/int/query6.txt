select ?tag count (*) #Q6
from <sib>
where {  
    {  select distinct ?fr
       from <sib>
       where {
           {%Person% snvoc:knows ?fr.} union {%Person% snvoc:knows ?fr2. ?fr2 snvoc:knows ?fr. filter (?fr != %Person%)}
       }
    } .
    ?post snvoc:hasCreator ?fr .
    ?post snvoc:hasTag ?tag .
    filter (?tag != %Tag%) .
    ?post snvoc:hasTag %Tag% .
}
group by ?tag
order by desc(2) ?tag
limit 10
