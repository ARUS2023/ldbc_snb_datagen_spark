select count(*)
where
  {
    {
      select ?s ?o
      where
        {
          ?s snvoc:knows ?o1.
          ?o1 snvoc:hasPerson ?o .
        }
    }
    option ( transitive,
             t_distinct,
             t_in(?s),
             t_out(?o),
             t_shortest_only,
             t_direction 3,
             t_step ('path_id') as ?path_no) .
    filter ( ?s = %Person1% ).
    filter ( ?o = %Person2% ).
    filter (?path_no = 0).
  }
