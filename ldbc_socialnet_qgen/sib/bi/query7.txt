select ?title ?volume  (100 *  count (*) / xsd:float (?volume))
where {
{select ?f (count (*) as ?volume) 
  where {?f sv:container_of ?post . ?post sv:has_tag ?tag . ?tag a %TagType% } 
  group by ?f having count (*) > 500}
.
 ?f sv:container_of ?post .
  ?post sv:has_tag %Name% .
?f sv:title ?title .
 } group by ?f ?title ?volume order by desc 3 limit 20
